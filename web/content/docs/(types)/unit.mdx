# Unit Type

The `Unit` type represents a function with no meaningful return value, similar to `void` in TypeScript.

## Overview

```ts
type Unit = undefined;

function returnsUnit(): Unit {
  console.log("Side effect");
  return undefined;
}
```

## Usage

### Unit Value

```ts
import { unit } from "@deessejs/functions";

const value: Unit = unit;
// undefined
```

### Functions Returning Unit

```ts
function logUser(user: User): Unit {
  console.log("User:", user);
  // No explicit return needed
}
```

### In Queries/Mutations

```ts
const logOperation = t.mutation({
  args: z.object({
    message: z.string(),
  }),
  handler: async (ctx, args) => {
    ctx.logger.info(args.message);
    return success(unit);
  },
});
```

## Examples

### Side Effects

```ts
const sendNotification = t.mutation({
  args: z.object({
    userId: z.number(),
    message: z.string(),
  }),
  handler: async (ctx, args) => {
    await ctx.notifications.send(args.userId, args.message);
    return success(unit);
  },
});

// Usage
const result = await api.sendNotification({
  userId: 1,
  message: "Hello!",
});

if (result.ok) {
  console.log("Notification sent");
  // result.value is undefined/unit
}
```

### Batch Operations

```ts
const processBatch = t.mutation({
  args: z.object({
    userIds: z.array(z.number()),
  }),
  handler: async (ctx, args) => {
    for (const userId of args.userIds) {
      await ctx.processUser(userId);
    }
    return success(unit);
  },
});
```

### Fire-and-Forget

```ts
const triggerBackgroundJob = t.mutation({
  args: z.object({
    jobId: z.string(),
  }),
  handler: async (ctx, args) => {
    // Queue the job but don't wait for it
    ctx.jobQueue.enqueue(args.jobId);
    return success(unit);
  },
});
```
